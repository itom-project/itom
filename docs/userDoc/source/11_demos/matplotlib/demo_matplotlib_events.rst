    def onclick(event):
        print(
            "%s click: button=%d, x=%d, y=%d, xdata=%f, ydata=%f"

    def onfigureenter(event):
        if event.x and event.y:
            print("onfigureenter: x=%f, y=%f, inaxes=%s" % (event.x, event.y, str(event.inaxes)))
