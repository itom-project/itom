    c = 0
    for row in range(0, 49):
        for col in range(0, 49):
            polygons[c, 0] = row * 50 + col
            polygons[c, 1] = (row + 1) * 50 + col
            polygons[c, 2] = row * 50 + 1 + col
            c += 1
    for row in range(0, 49):
